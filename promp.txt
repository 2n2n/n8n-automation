TYPE DEFINITION  (kitchenSheet.types.ts)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// Root type you must produce:
interface KitchenSheetPDF extends Omit<KitchenSheet, "menu"> {
  menu: MenuPDF;
}

interface KitchenSheet {
  updatedAt: string;               // e.g. "2/18/2026 2:01 pm"
  documentType: "KITCHEN SHEET";
  eventName: string;
  eventDate: string;               // e.g. "Sun, Mar 8, 2026"
  account: AccountInfo;
  cateringKitchen: CateringKitchenInfo;
  operations: OperationsInfo;
  dietaryRequirements: DietaryRequirements;
}

interface AccountInfo {
  account: string;
  accountManager: string;
  location: string | null;
  phone: string | null;
  bookingContact: string;
  bookingPhone: string;
  siteContact: string | null;
  sitePhone: string | null;
}

interface CateringKitchenInfo {
  date: string;
  setupType: "Buffet Style" | "Plated" | "Family Style" | string;
  eventType: "Lunch" | "Dinner" | "Breakfast" | "Snack" | string;
  guests: number;
}

interface OperationsInfo {
  date: string;
  offsiteLocation: string;
  offsiteAddress: string;
  serviceStartTime: string;        // e.g. "11:00 AM"
  serviceEndTime: string;          // e.g. "3:00 PM"
  arrivalTime: string;
  instructions: string;
}

interface MenuSectionPDFItem {
  quantity: number;                // Number of servings for this item (usually matches section qty)
  title: string;                   // Item name with dietary tags, e.g. "OREGANO ROASTED CHICKEN (gf+df)"
  ingredients: string;             // Ingredients line below the title. Empty string if none.
}

interface MenuSectionPDF {
  qty: number;                     // Section-level guest/serving count
  name: string;                    // Section name, e.g. "DODGERS LUNCH BUFFET"
  items: MenuSectionPDFItem[];
}

interface MenuPDF {
  date: string;
  serviceStart: string;
  serviceEnd: string;
  lkbTime: string;                 // Last Kitchen Briefing time
  sections: MenuSectionPDF[];      // One structured object per section block — see SECTION OBJECT FORMAT
}

interface DietaryRequirements {
  portionSizes: PortionSizes;
  specialMeals: SpecialMeal[];
}

interface PortionSizes {
  protein: string;                 // e.g. "8oz post-cook"
  starch: string;                  // e.g. "1 cup"
  vegetables: string;
}

interface SpecialMeal {
  qty: number;
  type: "WHOLE 30" | "SPECIALTY" | "CONCIERGE" | string;
  recipientName: string | null;
  restrictions: string[];
  description: string;
  deliveryTime: string | null;
  deliveredBy: string | null;
  servedHot: boolean;
  packaging: "togo box" | "plate" | null;
}

SECTION OBJECT FORMAT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Each entry in menu.sections is a structured object with qty, name, and items[].

  qty   — the number that precedes the section name in the PDF (e.g. 185)
  name  — the section name in UPPERCASE (e.g. "DODGERS LUNCH BUFFET")
  items — array of MenuSectionPDFItem objects, one per menu item line:
            - quantity:    same as the section qty
            - title:       item name with dietary tags in parentheses joined by "+"
                           Always UPPERCASE. e.g. "OREGANO ROASTED CHICKEN (gf+df)"
            - ingredients: the line directly below the item name in the PDF, comma-separated.
                           Use "" (empty string) if no ingredients line follows.

Example of a single section object:
  {
    "qty": 185,
    "name": "DODGERS LUNCH BUFFET",
    "items": [
      { "quantity": 185, "title": "OREGANO ROASTED CHICKEN (gf+df)", "ingredients": "caramelized lemon, fresh oregano" },
      { "quantity": 185, "title": "FRESH HERB ROASTED SALMON (gf+df)", "ingredients": "" },
      { "quantity": 185, "title": "MIXED ROASTED POTATOES (gf+df)", "ingredients": "peewee colored potatoes, fresh rosemary, thyme & sage" },
      { "quantity": 185, "title": "ROASTED ASPARAGUS (gf+df)", "ingredients": "" }
    ]
  }

RULES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. menu.sections is an array of structured objects — one object per section block. Each has qty, name, and items[].
2. Each section object's qty is the number before the section name in the PDF; name is the section title in UPPERCASE.
3. Each item in items[] uses MenuSectionPDFItem shape:
   - quantity:    same as the parent section's qty.
   - title:       item name always UPPERCASE, with dietary tags in parentheses joined by "+", e.g. "SALMON (gf+df)".
   - ingredients: the line directly below the item name in the PDF. Use "" if no ingredients line is present.
4. Use null for any field that is empty or not present in the document.
5. Dates must be copied exactly as they appear in the PDF — do not reformat.
6. Times must be in 12-hour format with AM/PM (e.g. "11:00 AM", "3:00 PM").
7. Phone numbers must include the country code prefix (e.g. "+14808782582").
8. portion_sizes values should include quantity + unit (e.g. "8oz post-cook", "1 cup").
9. Return ONLY the JSON object. No explanation, no markdown, no code fences.
